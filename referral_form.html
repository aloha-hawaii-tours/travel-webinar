's' : ''}`; const body = ` New referral submission received! 📅 Submitted: ${timestamp.toLocaleString('en-US', { timeZone: 'America/New_York' })} ET 👤 REFERRER Name: ${data.referrerName} Preferred Date: ${data.preferredDate} Timezone: ${data.timezone} Session Time: ${data.sessionTime} 👥 REFERRED FRIENDS (${friendCount}) ${friendList} --- View all submissions: ${SpreadsheetApp.getActiveSpreadsheet().getUrl()} `; MailApp.sendEmail({ to: YOUR_EMAIL, subject: subject, body: body, name: "Aloha Hawaii Referrals" }); } function doGet(e) { return ContentService.createTextOutput("Referral endpoint is active") .setMimeType(ContentService.MimeType.TEXT) .setHeaders({ 'Access-Control-Allow-Origin': '*' }); }
